3    from modelos.modelo_masa_puntual import ModeloMasaPuntual
4    from modelos.modelo_4dof import Modelo4DOF

7    class CalculadoraBalistica:
8      def __init__(self, modelo_seleccionado):
9          self.modelo_seleccionado = modelo_seleccionado  # 'masa_puntual' o '4dof'
10         self.configurar_modelo()
11   
12     def configurar_modelo(self):
13         if self.modelo_seleccionado == 'masa_puntual':
14             # Configuración para el modelos de masa puntual
15             # Constantes basadas en el paper del mortero
16             masa = 14.5  # kg
17             diametro = 0.12  # m
18             area = 0.0113  # m^2 (calculado o proporcionado)
19             cd = 0.295  # Coeficiente de arrastre base para proyectil de mortero
20             densidad_aire = 1.225  # kg/m^3 al nivel del mar
21   
22             self.modelo = ModeloMasaPuntual(masa, area, cd, densidad_aire)
23   
24         elif self.modelo_seleccionado == '4dof':
25             # Configuración para el modelos 4DOF
26             masa = 14.5  # kg
27             diametro = 0.12  # m
28             densidad_aire = 1.225  # kg/m^3 al nivel del mar
29             coeficientes = CoeficientesAerodinamicos()
30   
31             self.modelo = Modelo4DOF(masa, diametro, densidad_aire, coeficientes)
32   
33         else:
34             raise ValueError("Modelo seleccionado no es válido.")
35   
36     def establecer_parametros(self, angulo, velocidad_inicial, altura_inicial, densidad_aire):
37         # Actualizar densidad del aire en el modelos
38         self.modelo.densidad_aire = densidad_aire
39   
40         # Guardar parámetros para su uso posterior
41         self.angulo = angulo
42         self.velocidad_inicial = velocidad_inicial
43         self.altura_inicial = altura_inicial
44   
45     def calcular_trayectoria(self):
46         if self.modelo_seleccionado == 'masa_puntual':
47             self.modelo.calcular_trayectoria(self.angulo, self.velocidad_inicial, self.altura_inicial)
48         elif self.modelo_seleccionado == '4dof':
49             self.modelo.calcular_trayectoria(self.angulo, self.velocidad_inicial, self.altura_inicial)
50   
51         # Obtener resultados
52         self.tiempo_de_vuelo = self.modelo.tiempo_de_vuelo
53         self.alcance_maximo = self.modelo.alcance_maximo
54         self.altura_maxima = self.modelo.altura_maxima
55         self.Vx = self.modelo.Vx
56         self.Vy = self.modelo.Vy
57         self.X = self.modelo.X
58         self.Y = self.modelo.Y
59         self.tiempos = self.modelo.tiempos
60   
61     def graficar_trayectoria(self):
62         self.modelo.graficar_trayectoria()
63   
64     def exportar_a_csv(self, nombre_archivo):
65         # Exportar los resultados a un archivo CSV
66         import csv
67         with open(nombre_archivo, 'w', newline='') as csvfile:
68             campos = ['Tiempo (s)', 'Posición X (m)', 'Posición Y (m)', 'Vx (m/s)', 'Vy (m/s)']
69             escritor = csv.DictWriter(csvfile, fieldnames=campos)
70             escritor.writeheader()
71             for i in range(len(self.tiempos)):
72                 escritor.writerow({'Tiempo (s)': self.tiempos[i],
73                                    'Posición X (m)': self.X[i],
74                                    'Posición Y (m)': self.Y[i],
75                                    'Vx (m/s)': self.Vx[i],
76                                    'Vy (m/s)': self.Vy[i]})
