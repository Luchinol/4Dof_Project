3    import tkinter as tk
4    from tkinter import filedialog, messagebox
5    from calculadora.calculadora_balistica import CalculadoraBalistica
6    
7    class AplicacionGUI(tk.Tk):
8      def __init__(self):
9          super().__init__()
10         self.title("Calculadora Balística")
11         self.geometry("400x400")
12         self.modelo_seleccionado = tk.StringVar(value='masa_puntual')
13         self.crear_widgets()
14   
15     def crear_widgets(self):
16         # Selección del modelos
17         tk.Label(self, text="Seleccione el Modelo:").grid(row=0, column=0, sticky='e')
18         tk.Radiobutton(self, text="Masa Puntual", variable=self.modelo_seleccionado,
19                        value='masa_puntual').grid(row=0, column=1, sticky='w')
20         tk.Radiobutton(self, text="4DOF", variable=self.modelo_seleccionado,
21                        value='4dof').grid(row=1, column=1, sticky='w')
22   
23         # Etiquetas y campos de entrada
24         tk.Label(self, text="Ángulo de Ataque/Lanzamiento (grados):").grid(row=2, column=0, sticky='e')
25         self.entrada_angulo = tk.Entry(self)
26         self.entrada_angulo.grid(row=2, column=1)
27   
28         tk.Label(self, text="Velocidad Inicial (m/s):").grid(row=3, column=0, sticky='e')
29         self.entrada_velocidad = tk.Entry(self)
30         self.entrada_velocidad.grid(row=3, column=1)
31   
32         tk.Label(self, text="Altura Inicial (m):").grid(row=4, column=0, sticky='e')
33         self.entrada_altura = tk.Entry(self)
34         self.entrada_altura.grid(row=4, column=1)
35   
36         tk.Label(self, text="Densidad del Aire (kg/m³):").grid(row=5, column=0, sticky='e')
37         self.entrada_densidad = tk.Entry(self)
38         self.entrada_densidad.grid(row=5, column=1)
39         self.entrada_densidad.insert(0, "1.225")  # Valor por defecto al nivel del mar
40   
41         # Etiquetas de salida
42         self.etiqueta_tiempo_vuelo = tk.Label(self, text="Tiempo de Vuelo: N/A")
43         self.etiqueta_tiempo_vuelo.grid(row=7, column=0, columnspan=2)
44   
45         self.etiqueta_alcance_maximo = tk.Label(self, text="Alcance Máximo: N/A")
46         self.etiqueta_alcance_maximo.grid(row=8, column=0, columnspan=2)
47   
48         self.etiqueta_altura_maxima = tk.Label(self, text="Altura Máxima: N/A")
49         self.etiqueta_altura_maxima.grid(row=9, column=0, columnspan=2)
50   
51         # Botones
52         tk.Button(self, text="Calcular", command=self.calcular).grid(row=6, column=0)
53         tk.Button(self, text="Reiniciar", command=self.reiniciar).grid(row=6, column=1)
54         tk.Button(self, text="Graficar Trayectoria", command=self.graficar_trayectoria).grid(row=10, column=0)
55         tk.Button(self, text="Exportar a CSV", command=self.exportar_csv).grid(row=10, column=1)
56   
57     def calcular(self):
58         try:
59             angulo = float(self.entrada_angulo.get())
60             velocidad = float(self.entrada_velocidad.get())
61             altura = float(self.entrada_altura.get())
62             densidad = float(self.entrada_densidad.get())
63             modelo = self.modelo_seleccionado.get()
64   
65             self.calculadora = CalculadoraBalistica(modelo)
66             self.calculadora.establecer_parametros(angulo, velocidad, altura, densidad)
67             self.calculadora.calcular_trayectoria()
68   
69             # Actualizar etiquetas de salida
70             self.etiqueta_tiempo_vuelo.config(text=f"Tiempo de Vuelo: {self.calculadora.tiempo_de_vuelo:.2f} s")
71             self.etiqueta_alcance_maximo.config(text=f"Alcance Máximo: {self.calculadora.alcance_maximo:.2f} m")
72             self.etiqueta_altura_maxima.config(text=f"Altura Máxima: {self.calculadora.altura_maxima:.2f} m")
73   
74             messagebox.showinfo("Cálculo Completo", "El cálculo de la trayectoria se completó exitosamente.")
75   
76         except ValueError:
77             messagebox.showerror("Error de Entrada", "Por favor, ingresa valores numéricos válidos.")
78   
79     def reiniciar(self):
80         # Limpiar campos y etiquetas
81         self.entrada_angulo.delete(0, tk.END)
82         self.entrada_velocidad.delete(0, tk.END)
83         self.entrada_altura.delete(0, tk.END)
84         self.entrada_densidad.delete(0, tk.END)
85         self.entrada_densidad.insert(0, "1.225")  # Valor por defecto
86         self.etiqueta_tiempo_vuelo.config(text="Tiempo de Vuelo: N/A")
87         self.etiqueta_alcance_maximo.config(text="Alcance Máximo: N/A")
88         self.etiqueta_altura_maxima.config(text="Altura Máxima: N/A")
89         self.modelo_seleccionado.set('masa_puntual')
90         if hasattr(self, 'calculadora'):
91             del self.calculadora
92   
93     def graficar_trayectoria(self):
94         if hasattr(self, 'calculadora') and self.calculadora.X:
95             self.calculadora.graficar_trayectoria()
96         else:
97             messagebox.showwarning("Sin Datos", "Por favor, realiza un cálculo primero.")
98   
99     def exportar_csv(self):
100        if hasattr(self, 'calculadora') and self.calculadora.tiempos:
101            archivo = filedialog.asksaveasfilename(defaultextension='.csv',
102                                                   filetypes=[("Archivos CSV", "*.csv")])
103            if archivo:
104                self.calculadora.exportar_a_csv(archivo)
105                messagebox.showinfo("Exportación Completa", f"Datos exportados a {archivo}")
106        else:
107            messagebox.showwarning("Sin Datos", "Por favor, realiza un cálculo primero.")
